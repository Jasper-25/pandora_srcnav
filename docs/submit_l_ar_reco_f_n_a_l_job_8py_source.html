67def run(pars):
68
69 print('Base job directory = {0}; label = {1}, sample = {2}'.format(pars.baseJob, pars.label, pars.sample))
70 print('jobName = {0}, inFileName = {1}, events = {2} to {3}'.format(pars.jobName, pars.inFileName,
71 pars.startEvt, pars.endEvt))
72
73 print('OSVer = {0}'.format(pars.OSVer))
74
75 # Run directory for the job
76 print('JobDir = {0}'.format(pars.jobDir))
77 if not os.path.exists(pars.jobDir):
78 print('Creating {0}'.format(pars.jobDir))
79 os.makedirs(pars.jobDir)
80 os.chmod(pars.jobDir, 0o0755)
81
82 # Create tar file containing Pandora release and copy it to scratch area
83 if not os.path.exists(pars.pandoraTarFile):
84 print('Creating {0}'.format(pars.pandoraTarFile))
85 # List of required install packages
86 pgkList = ' '.join([str(pkg) for pkg in pars.pandoraPkgs])
87 tarCmd = 'tar -czf {0} -C {1} {2}'.format(pars.pandoraTarFile, pars.myTestArea, pgkList)
88 print('tarCmd = {0}'.format(tarCmd))
89 os.system(tarCmd)
90 else:
91 # Tar file has already been created
92 print('Using tarball {0}'.format(pars.pandoraTarFile))
93
94 # Now create the script containing the job commands
95 jobScript = createJobScript(pars)
96
97 # Finally submit the job; for tests, use --timeout=Xm to stop after X minutes.
98 # Majority of jobs should finish within 8 hours and use a max memory of 2GB
99 logFile = '{0}/submitJob.log'.format(pars.jobDir)
100 jobCmd = 'jobsub_submit.py -N 1 --resource-provides=usage_model=OPPORTUNISTIC ' \
101 '--expected-lifetime={0} --OS={1} --group=dune --memory={2} -L {3} ' \
102 'file://{4}'.format(pars.runtime, pars.OSVer, pars.memory, logFile, jobScript)
103 print('jobCmd = {0}'.format(jobCmd))
104
105 os.system(jobCmd)
106 print('DONE\n')

109def createJobScript(pars):
110
111 jobScript = '{0}/{1}_{2}.sh'.format(pars.jobDir, pars.sample, pars.jobName)
112 print('Creating job script {0}'.format(jobScript))
113 if os.path.exists(jobScript):
114 os.remove(jobScript)
115 jobFile = open(jobScript, 'w')
116
117 # Setup job environment
118 jobFile.write('#!/bin/bash\n')
119 jobFile.write('date\n')
120 jobFile.write('source /cvmfs/dune.opensciencegrid.org/products/dune/setup_dune.sh\n')
121 jobFile.write('setup ifdhc\n')
122
123 # For libxxhash.so
124 jobFile.write('setup dune_oslibs v1_0_0\n')
125 # Compiler
126 jobFile.write('setup gcc v8_2_0\n')
127 # ROOT
128 jobFile.write('setup root v6_18_04d -q e19:prof\n')
129 # Geant4 (for edep-sim)
130 jobFile.write('setup geant4 v4_10_6_p01c -q e19:prof\n')
131
132 # Limit ifdh copying attempts to avoid stalled jobs
133 jobFile.write('export IFDH_CP_MAXRETRIES=0\n')
134
135 # Print working directory
136 jobFile.write('echo Batch dir is {0}\n'.format(pars.batchDir))
137
138 # Copy Pandora release tarball from pnfs scratch area to batch machine local area
139 batchTarFile = '{0}/{1}'.format(pars.batchDir, pars.pandoraTarName)
140 jobFile.write('ifdh cp {0} {1}\n'.format(pars.pandoraTarFile, batchTarFile))
141 jobFile.write('tar -xzf {0} -C {1}\n\n'.format(batchTarFile, pars.batchDir))
142
143 # Specify edep-sim header file location for LArReco executable (for ROOT-based shared-libraries)
144 edepSimHDir = '$_CONDOR_SCRATCH_DIR/edep-sim/install/include/EDepSim'
145 jobFile.write('export ROOT_INCLUDE_PATH={0}:$ROOT_INCLUDE_PATH\n'.format(edepSimHDir))
146
147 # LD_LIBRARY_PATH for shared libraries needed for LArReco executable
148 libPath1 = '$_CONDOR_SCRATCH_DIR/PandoraSDK/lib'
149 libPath2 = '$_CONDOR_SCRATCH_DIR/LArContent/lib'
150 libPath3 = '$_CONDOR_SCRATCH_DIR/PandoraMonitoring/lib'
151 libPath4 = '$_CONDOR_SCRATCH_DIR/edep-sim/install/lib'
152 jobFile.write('export LD_LIBRARY_PATH={0}:{1}:{2}:{3}:$LD_LIBRARY_PATH\n'.format(libPath1, libPath2,
153 libPath3, libPath4))
154 # Location of Pandora algorithm parameter files
155 FWSPath1 = '$_CONDOR_SCRATCH_DIR/LArMachineLearningData'
156 FWSPath2 = '$_CONDOR_SCRATCH_DIR/LArReco/settings'
157 jobFile.write('export FW_SEARCH_PATH={0}:{1}\n'.format(FWSPath1, FWSPath2))
158
159 # Copy edep-sim rooTracker inputFile to local batch dir
160 jobFile.write('ifdh cp -D {0} $_CONDOR_SCRATCH_DIR\n'.format(pars.inFileName))
161
162 # Print directory contents to check presence of directories and inputFile
163 jobFile.write('ls -lrta {0}\n\n'.format(pars.batchDir))
164
165 # LArReco run command; "-N" prints event numbers. Use "-h" for help options.
166 # Can also add extra arguments: "-c X" for MIP calib cut = X (default = 0.3) and
167 # "-m X" to set max number of merged voxels per event (default = -1, all voxels)
168 fileName = pars.inFileName.split('/')[-1]
169 runCmd = './{0} -i {1} -j {2} -N -s {3} -n {4} -r {5} ' \
170 '-e {6}\n'.format(pars.recoExe, pars.recoXml, pars.projection,
171 pars.startEvt, pars.NEvts, pars.recoOpt, fileName)
172
173 jobFile.write('cd {0}\n'.format(pars.batchDir))
174 jobFile.write(runCmd)
175
176 # List all output to make sure everything was generated OK
177 jobFile.write('ls -lrt {0}\n\n'.format(pars.batchDir))
178
179 # Copy Validation.root file from batch node to pnfs user scratch area.
180 # Note that the file will not be copied if it already exists!
181 jobFile.write('ifdh cp -D Validation.root {0}\n'.format(pars.jobDir))
182
183 # End the job
184 jobFile.write('echo Copied Validation.root to {0}\n'.format(pars.jobDir))
185 jobFile.write('date\n')
186 jobFile.write('exit 0\n')
187
188 jobFile.close()
189
190 # Make job script executable
191 os.chmod(jobScript, 0o0644)
192
193 return jobScript
